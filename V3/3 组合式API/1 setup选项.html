setup选项的写法和执行机制 1.执行时机，比beforeCreate还要早 2.setup函数中，获取不到this(this还是undefined) export default { setup () { console.log('setup函数', this) // undefined }, beforeCreate() { console.log('beforeCreate函数') } }
3.数据和函数，需要在setup最后return，才能模版中应用 export default { setup () { // 数据 const message = 'hello Vue3' // 函数 const logMessage = () => { console.log(message) } return { message, logMessage } } } {{ message }}
